Deploy Edge Function `send-report` with JWT verification DISABLED (manual auth check like generate-pdf).

**Function signature:**
```
POST /functions/v1/send-report
Headers: Authorization: Bearer <user_jwt>
Body: {
  visit_id: string,
  document_id: string,
  recipients: string[]  // email addresses
}
Response: { sent_to: number, document_id: string }
```

**Implementation:**

```typescript
import "jsr:@supabase/functions-js/edge-runtime.d.ts";
import { createClient } from "jsr:@supabase/supabase-js@2";

Deno.serve(async (req: Request) => {
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders });
  }

  try {
    // 1. Auth check
    const authHeader = req.headers.get('Authorization');
    if (!authHeader) return errorResponse('Missing authorization', 401);

    const supabase = createClient(
      Deno.env.get('SUPABASE_URL')!,
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!,
    );

    // Verify user
    const token = authHeader.replace('Bearer ', '');
    const { data: { user }, error: authErr } = await supabase.auth.getUser(token);
    if (authErr || !user) return errorResponse('Unauthorized', 401);

    // 2. Parse body
    const { visit_id, document_id, recipients } = await req.json();
    if (!visit_id || !document_id || !recipients?.length) {
      return errorResponse('Missing visit_id, document_id, or recipients', 400);
    }

    // 3. Fetch document
    const { data: doc, error: docErr } = await supabase
      .from('generated_documents')
      .select('*')
      .eq('id', document_id)
      .single();
    if (docErr || !doc) return errorResponse('Document not found', 404);

    // 4. Fetch visit + project info for email template
    const { data: visit } = await supabase
      .from('visits')
      .select('*, projects(name, address)')
      .eq('id', visit_id)
      .single();
    if (!visit) return errorResponse('Visit not found', 404);

    const projectName = (visit as any).projects?.name || 'Projet';
    const visitDate = visit.date;

    // 5. Send email via Resend
    const RESEND_API_KEY = Deno.env.get('RESEND_API_KEY');
    if (!RESEND_API_KEY) return errorResponse('RESEND_API_KEY not configured', 500);

    const emailHtml = `
      <div style="font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif; max-width: 600px; margin: 0 auto; padding: 32px;">
        <h2 style="color: #1E293B; margin-bottom: 4px;">Compte Rendu de Visite</h2>
        <p style="color: #64748B; margin-top: 0;">${projectName} — ${visitDate}</p>
        <hr style="border: none; border-top: 1px solid #E2E8F0; margin: 24px 0;" />
        <p style="color: #334155;">
          Veuillez trouver ci-joint le compte rendu de visite pour le projet <strong>${projectName}</strong>.
        </p>
        <a href="${doc.file_url}" style="display: inline-block; background: #F59E0B; color: #1a1a2e; padding: 12px 24px; border-radius: 8px; text-decoration: none; font-weight: 600; margin: 16px 0;">
          📄 Télécharger le PDF (v${doc.version})
        </a>
        <hr style="border: none; border-top: 1px solid #E2E8F0; margin: 24px 0;" />
        <p style="color: #94A3B8; font-size: 12px;">
          Envoyé via Raporitfy — Comptes Rendus de Chantier
        </p>
      </div>
    `;

    const resendRes = await fetch('https://api.resend.com/emails', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${RESEND_API_KEY}`,
      },
      body: JSON.stringify({
        from: 'Raporitfy <cr@raporitfy.com>',
        to: recipients,
        subject: `CR Visite — ${projectName} — ${visitDate}`,
        html: emailHtml,
        attachments: [
          {
            filename: `CR_${projectName.replace(/\s+/g, '_')}_v${doc.version}.pdf`,
            path: doc.file_url,
          },
        ],
      }),
    });

    if (!resendRes.ok) {
      const err = await resendRes.text();
      console.error('Resend error:', err);
      return errorResponse(`Email send failed: ${err}`, 500);
    }

    // 6. Update generated_documents
    await supabase
      .from('generated_documents')
      .update({
        sent_at: new Date().toISOString(),
        recipients: recipients,
      })
      .eq('id', document_id);

    // 7. Update visit status to 'diffuse'
    await supabase
      .from('visits')
      .update({ status: 'diffuse' })
      .eq('id', visit_id);

    return new Response(
      JSON.stringify({ sent_to: recipients.length, document_id }),
      { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
    );

  } catch (err) {
    console.error('send-report error:', err);
    return errorResponse('Internal error', 500);
  }
});

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
};

function errorResponse(message: string, status: number) {
  return new Response(
    JSON.stringify({ error: message }),
    { status, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
  );
}
```

**NOTE on `from` address:** Resend requires a verified domain. For MVP, use `onboarding@resend.dev` during development (Resend's test sender). Replace with branded domain later.

**Secrets to set:**
```bash
supabase secrets set RESEND_API_KEY=re_xxxxx
```
