</read_first>

<action>
The RESEND_API_KEY must be set before the function can send emails.

```bash
# User must run this manually or via Supabase dashboard:
supabase secrets set RESEND_API_KEY=re_xxxxx
```

**NOTE:** User needs a Resend account (resend.com — free tier = 100 emails/day). The API key is found at resend.com/api-keys.

For MVP, we use `onboarding@resend.dev` as the sender (no domain verification required).
Production will require adding `raporitfy.com` as a verified domain in Resend.

<action>
Deploy Edge Function `send-report` via MCP `deploy_edge_function` with `verify_jwt: true`.

**Function signature:**
```
POST /functions/v1/send-report
Headers: Authorization: Bearer <user_jwt>
Body: {
  visit_id: string,
  document_id: string,
  recipients: string[]
}
Response: { sent_to: number, document_id: string }
```

**Full implementation (`index.ts`):**

```typescript
import "jsr:@supabase/functions-js/edge-runtime.d.ts";
import { createClient } from "jsr:@supabase/supabase-js@2";

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
};

function errorResponse(message: string, status: number) {
  return new Response(
    JSON.stringify({ error: message }),
    { status, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
  );
}

Deno.serve(async (req: Request) => {
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders });
  }

  try {
    const supabase = createClient(
      Deno.env.get('SUPABASE_URL')!,
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!,
    );

    // Auth: verify user (gateway already checked JWT with verify_jwt: true)
    const authHeader = req.headers.get('Authorization');
    if (!authHeader) return errorResponse('Missing authorization', 401);
    const token = authHeader.replace('Bearer ', '');
    const { data: { user }, error: authErr } = await supabase.auth.getUser(token);
    if (authErr || !user) return errorResponse('Unauthorized', 401);

    // Parse body
    const { visit_id, document_id, recipients } = await req.json();
    if (!visit_id || !document_id || !recipients?.length) {
      return errorResponse('Missing visit_id, document_id, or recipients', 400);
    }

    // Fetch document
    const { data: doc, error: docErr } = await supabase
      .from('generated_documents')
      .select('*')
      .eq('id', document_id)
      .single();
    if (docErr || !doc) return errorResponse('Document not found', 404);

    // Fetch visit + project
    const { data: visit } = await supabase
      .from('visits')
      .select('*')
      .eq('id', visit_id)
      .single();
    if (!visit) return errorResponse('Visit not found', 404);

    const { data: project } = await supabase
      .from('projects')
      .select('name, address')
      .eq('id', visit.project_id)
      .single();

    const projectName = project?.name || 'Projet';
    const visitDate = visit.date;

    // Send email via Resend
    const RESEND_API_KEY = Deno.env.get('RESEND_API_KEY');
    if (!RESEND_API_KEY) return errorResponse('RESEND_API_KEY not configured', 500);

    // GAP-G FIX: use onboarding@resend.dev for MVP (no domain verification needed)
    const fromAddress = 'Raporitfy <onboarding@resend.dev>';

    const emailHtml = `
<!DOCTYPE html>
<html>
<body style="margin:0; padding:0; background:#f8fafc; font-family:-apple-system,BlinkMacSystemFont,'Segoe UI',Roboto,sans-serif;">
  <div style="max-width:600px; margin:0 auto; padding:40px 24px;">
    <div style="background:#1a1a2e; border-radius:12px; padding:32px; color:#ffffff;">
      <h1 style="margin:0 0 4px; font-size:20px; color:#F59E0B;">
        Compte Rendu de Chantier N°${doc.version}
      </h1>
      <p style="margin:0 0 24px; color:#94A3B8; font-size:14px;">
        ${projectName} — ${visitDate}
      </p>
      <hr style="border:none; border-top:1px solid #334155; margin:0 0 24px;" />
      <p style="color:#E2E8F0; font-size:14px; line-height:1.6;">
        Veuillez trouver ci-joint le compte rendu de visite de chantier pour le projet
        <strong style="color:#ffffff;">${projectName}</strong>${project?.address ? ` (${project.address})` : ''}.
      </p>
      <a href="${doc.file_url}"
         style="display:inline-block; background:#F59E0B; color:#1a1a2e; padding:14px 28px;
                border-radius:8px; text-decoration:none; font-weight:700; font-size:14px; margin:20px 0;">
        📄 Télécharger le CR (PDF)
      </a>
      <p style="color:#64748B; font-size:12px; margin-top:24px;">
        Ce document est également disponible en pièce jointe de cet email.
      </p>
    </div>
    <p style="text-align:center; color:#94A3B8; font-size:11px; margin-top:24px;">
      Envoyé via <strong>Raporitfy</strong> — Comptes Rendus de Chantier
    </p>
  </div>
</body>
</html>`;

    const resendRes = await fetch('https://api.resend.com/emails', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${RESEND_API_KEY}`,
      },
      body: JSON.stringify({
        from: fromAddress,
        to: recipients,
        subject: `CR N°${doc.version} — ${projectName} — ${visitDate}`,
        html: emailHtml,
        attachments: [
          {
            filename: `CR_${projectName.replace(/[^a-zA-Z0-9]/g, '_')}_N${doc.version}.pdf`,
            path: doc.file_url,
          },
        ],
      }),
    });

    if (!resendRes.ok) {
      const errText = await resendRes.text();
      console.error('Resend API error:', errText);
      return errorResponse(`Erreur email: ${errText}`, 500);
    }

    // Update generated_documents: mark as sent
    await supabase
      .from('generated_documents')
      .update({
        sent_at: new Date().toISOString(),
        recipients: recipients,
      })
      .eq('id', document_id);

    // Update visit status to 'diffuse'
    await supabase
      .from('visits')
      .update({ status: 'diffuse' })
      .eq('id', visit_id);

    return new Response(
      JSON.stringify({ sent_to: recipients.length, document_id }),
      { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
    );

  } catch (err: any) {
    console.error('send-report error:', err);
    return errorResponse(err.message || 'Internal error', 500);
  }
});
```
