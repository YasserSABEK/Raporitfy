```typescript
import "jsr:@supabase/functions-js/edge-runtime.d.ts";
import { createClient } from "jsr:@supabase/supabase-js@2";
// GAP-A FIX: use npm: specifier, NOT skypack CDN
import { PDFDocument, rgb, StandardFonts } from "npm:pdf-lib@1.17.1";

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
};

Deno.serve(async (req: Request) => {
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders });
  }

  try {
    const SUPABASE_URL = Deno.env.get('SUPABASE_URL')!;
    const supabase = createClient(
      SUPABASE_URL,
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!,
    );

    // Auth: get user from JWT (verify_jwt: true means gateway already validated)
    const authHeader = req.headers.get('Authorization');
    if (!authHeader) return errorResponse('Missing authorization', 401);
    const token = authHeader.replace('Bearer ', '');
    const { data: { user }, error: authErr } = await supabase.auth.getUser(token);
    if (authErr || !user) return errorResponse('Unauthorized', 401);

    // Parse body
    const { visit_id } = await req.json();
    if (!visit_id) return errorResponse('Missing visit_id', 400);

    // ===== FETCH ALL DATA =====
    // 1. Visit + Project
    const { data: visit, error: visitErr } = await supabase
      .from('visits')
      .select('*')
      .eq('id', visit_id)
      .single();
    if (visitErr || !visit) return errorResponse('Visit not found', 404);

    const { data: project } = await supabase
      .from('projects')
      .select('*')
      .eq('id', visit.project_id)
      .single();

    // 2. Observations + evidence
    const { data: observations } = await supabase
      .from('observations')
      .select('*, evidence(id, type, file_url)')
      .eq('visit_id', visit_id)
      .order('created_at', { ascending: true });

    // 3. Decisions
    const { data: decisions } = await supabase
      .from('decisions')
      .select('*')
      .eq('visit_id', visit_id)
      .order('created_at', { ascending: true });

    // 4. Actions (this visit)
    const { data: actions } = await supabase
      .from('action_items')
      .select('*')
      .eq('visit_id', visit_id)
      .order('created_at', { ascending: true });

    // 5. Carry-forward actions (open actions from OTHER visits in same project)
    const { data: projectVisits } = await supabase
      .from('visits')
      .select('id')
      .eq('project_id', visit.project_id)
      .neq('id', visit_id);
    const otherVisitIds = (projectVisits || []).map((v: any) => v.id);
    let carryForward: any[] = [];
    if (otherVisitIds.length > 0) {
      const { data: cfActions } = await supabase
        .from('action_items')
        .select('*')
        .in('visit_id', otherVisitIds)
        .in('status', ['ouverte', 'en_cours']);
      carryForward = cfActions || [];
    }

    // 6. Recipients (for diffusion footer)
    const { data: recipients } = await supabase
      .from('project_recipients')
      .select('email, name')
      .eq('project_id', visit.project_id);

    // 7. User profile (MOE name)
    const { data: profile } = await supabase
      .from('profiles')
      .select('full_name, email')
      .eq('id', user.id)
      .single();

    // ===== VERSION CALCULATION =====
    const { data: existing } = await supabase
      .from('generated_documents')
      .select('version')
      .eq('visit_id', visit_id)
      .order('version', { ascending: false })
      .limit(1);
    const nextVersion = existing && existing.length > 0 ? existing[0].version + 1 : 1;

    // ===== BUILD PDF (French CR de Chantier Standard) =====
    const pdfDoc = await PDFDocument.create();
    const fontRegular = await pdfDoc.embedFont(StandardFonts.Helvetica);
    const fontBold = await pdfDoc.embedFont(StandardFonts.HelveticaBold);

    const PAGE_W = 595.28; // A4
    const PAGE_H = 841.89;
    const MARGIN = 50;
    const CONTENT_W = PAGE_W - 2 * MARGIN;
    let page = pdfDoc.addPage([PAGE_W, PAGE_H]);
    let y = PAGE_H - MARGIN;

    const DARK = rgb(0.1, 0.1, 0.15);
    const GRAY = rgb(0.4, 0.4, 0.45);
    const AMBER = rgb(0.96, 0.62, 0.04);
    const LINE_H = 14;
    const SECTION_GAP = 20;

    // Helper: add text with word wrap
    function drawText(text: string, x: number, options: { font?: any; size?: number; color?: any; maxWidth?: number } = {}) {
      const font = options.font || fontRegular;
      const size = options.size || 10;
      const color = options.color || DARK;
      const maxW = options.maxWidth || CONTENT_W;

      // Simple word wrap
      const words = (text || '').split(' ');
      let line = '';
      for (const word of words) {
        const test = line ? `${line} ${word}` : word;
        const width = font.widthOfTextAtSize(test, size);
        if (width > maxW && line) {
          if (y < MARGIN + 40) { page = pdfDoc.addPage([PAGE_W, PAGE_H]); y = PAGE_H - MARGIN; }
          page.drawText(line, { x, y, size, font, color });
          y -= LINE_H;
          line = word;
        } else {
          line = test;
        }
      }
      if (line) {
        if (y < MARGIN + 40) { page = pdfDoc.addPage([PAGE_W, PAGE_H]); y = PAGE_H - MARGIN; }
        page.drawText(line, { x, y, size, font, color });
        y -= LINE_H;
      }
    }

    function drawSectionHeader(title: string) {
      y -= SECTION_GAP;
      if (y < MARGIN + 60) { page = pdfDoc.addPage([PAGE_W, PAGE_H]); y = PAGE_H - MARGIN; }
      // Amber line
      page.drawRectangle({ x: MARGIN, y: y + 2, width: CONTENT_W, height: 2, color: AMBER });
      y -= 4;
      drawText(title, MARGIN, { font: fontBold, size: 12, color: DARK });
      y -= 4;
    }

    function drawHR() {
      page.drawLine({ start: { x: MARGIN, y }, end: { x: PAGE_W - MARGIN, y }, thickness: 0.5, color: rgb(0.85, 0.85, 0.85) });
      y -= 8;
    }

    // ===== 1. HEADER — "COMPTE RENDU DE CHANTIER N°X" =====
    drawText('COMPTE RENDU DE CHANTIER', MARGIN, { font: fontBold, size: 18, color: DARK });
    drawText(`N° ${nextVersion}`, MARGIN, { font: fontBold, size: 14, color: AMBER });
    y -= 8;

    // ===== 2. BLOC IDENTIFICATION =====
    const projectName = project?.name || 'Projet';
    const projectAddr = project?.address || '';
    drawText(`Chantier : ${projectName}`, MARGIN, { font: fontBold, size: 11 });
    if (projectAddr) drawText(`Adresse : ${projectAddr}`, MARGIN, { size: 10, color: GRAY });
    drawText(`Maître d'œuvre : ${profile?.full_name || user.email}`, MARGIN, { size: 10 });
    drawText(`Date de visite : ${visit.date}`, MARGIN, { size: 10 });
    drawText(`Type : ${visit.type || 'chantier'}`, MARGIN, { size: 10 });
    if (visit.weather) drawText(`Météo : ${visit.weather}`, MARGIN, { size: 10 });
    drawHR();

    // ===== 3. PARTICIPANTS =====
    if (visit.participants && visit.participants.length > 0) {
      drawSectionHeader('PARTICIPANTS');
      drawText(`Présents : ${visit.participants.join(', ')}`, MARGIN, { size: 10 });
    }

    // ===== 4. SUIVI DES ACTIONS ANTÉRIEURES (carry-forward) =====
    if (carryForward.length > 0) {
      drawSectionHeader('SUIVI DES ACTIONS ANTÉRIEURES');
      for (const cf of carryForward) {
        const statusLabel = cf.status === 'en_cours' ? 'En cours' : 'Ouverte';
        drawText(`• [${statusLabel}] ${cf.description} — ${cf.owner}${cf.deadline ? ` (éch. ${cf.deadline})` : ''}`, MARGIN, { size: 9 });
      }
    }

    // ===== 5. OBSERVATIONS PAR LOT =====
    if (observations && observations.length > 0) {
      drawSectionHeader('OBSERVATIONS');
      // Group by lot
      const byLot: Record<string, any[]> = {};
      for (const obs of observations) {
        const lot = obs.lot || 'Divers';
        if (!byLot[lot]) byLot[lot] = [];
        byLot[lot].push(obs);
      }
      for (const [lot, obsList] of Object.entries(byLot)) {
        drawText(`Lot : ${lot}`, MARGIN, { font: fontBold, size: 10 });
        for (let i = 0; i < obsList.length; i++) {
          const obs = obsList[i];
          const sevLabel = obs.severity === 'critique' ? '🔴' : obs.severity === 'majeur' ? '🟠' : '🟡';
          drawText(`  ${i + 1}. ${sevLabel} [${obs.zone}] ${obs.description}`, MARGIN + 10, { size: 9, maxWidth: CONTENT_W - 10 });

          // GAP-B FIX: photo URLs are PATHS — build full URL
          const evidenceList = Array.isArray(obs.evidence) ? obs.evidence : [];
          const photos = evidenceList.filter((e: any) => e.type === 'photo' && e.file_url);
          if (photos.length > 0) {
            for (const photo of photos) {
              try {
                const photoUrl = `${SUPABASE_URL}/storage/v1/object/public/evidence-photos/${photo.file_url}`;
                const imgRes = await fetch(photoUrl);
                if (imgRes.ok) {
                  const imgBytes = new Uint8Array(await imgRes.arrayBuffer());
                  let embedded;
                  const ext = (photo.file_url as string).toLowerCase();
                  if (ext.endsWith('.png')) {
                    embedded = await pdfDoc.embedPng(imgBytes);
                  } else {
                    embedded = await pdfDoc.embedJpg(imgBytes);
                  }
                  // Scale to max 180x120
                  const scale = Math.min(180 / embedded.width, 120 / embedded.height, 1);
                  const w = embedded.width * scale;
                  const h = embedded.height * scale;
                  if (y - h < MARGIN + 40) { page = pdfDoc.addPage([PAGE_W, PAGE_H]); y = PAGE_H - MARGIN; }
                  y -= h + 4;
                  page.drawImage(embedded, { x: MARGIN + 20, y, width: w, height: h });
                  y -= 8;
                }
              } catch (e) {
                // Photo fetch failed — skip silently
                console.warn(`Photo embed failed: ${photo.file_url}`, e);
              }
            }
          }
        }
        y -= 4;
      }
    }

    // ===== 6. DÉCISIONS PRISES =====
    if (decisions && decisions.length > 0) {
      drawSectionHeader('DÉCISIONS PRISES');
      for (const dec of decisions) {
        const scope = dec.scope === 'contractuel' ? '[Contractuel]' : dec.scope === 'chantier_global' ? '[Global]' : '[Lot]';
        const validated = dec.validated ? ' ✓ Validée' : '';
        drawText(`• ${scope} ${dec.content} — ${dec.author}${validated}`, MARGIN, { size: 9, maxWidth: CONTENT_W });
      }
    }

    // ===== 7. ACTIONS À MENER =====
    if (actions && actions.length > 0) {
      drawSectionHeader('ACTIONS À MENER');
      // Table header
      drawText('  Description                              Responsable       Échéance     Priorité', MARGIN, { font: fontBold, size: 8, color: GRAY });
      y -= 2;
      for (const act of actions) {
        const pr = act.priority === 'urgente' ? '🔴 Urgente' : act.priority === 'haute' ? '🟠 Haute' : act.priority === 'moyenne' ? '🟡 Moyenne' : '⚪ Basse';
        const desc = (act.description || '').substring(0, 40);
        const owner = (act.owner || '').substring(0, 18);
        const dl = act.deadline || '—';
        drawText(`  ${desc.padEnd(42)} ${owner.padEnd(18)} ${dl.padEnd(12)} ${pr}`, MARGIN, { size: 8 });
      }
    }

    // ===== 8. DIFFUSION =====
    if (recipients && recipients.length > 0) {
      drawSectionHeader('DIFFUSION');
      for (const r of recipients) {
        drawText(`• ${r.name || r.email} — ${r.email}`, MARGIN, { size: 9, color: GRAY });
      }
    }

    // ===== FOOTER on every page =====
    const pages = pdfDoc.getPages();
    for (let i = 0; i < pages.length; i++) {
      const p = pages[i];
      p.drawText(`CR N°${nextVersion} — ${projectName}`, { x: MARGIN, y: 25, size: 8, font: fontRegular, color: GRAY });
      p.drawText(`Page ${i + 1}/${pages.length}`, { x: PAGE_W - MARGIN - 60, y: 25, size: 8, font: fontRegular, color: GRAY });
      p.drawText('Raporitfy', { x: PAGE_W / 2 - 20, y: 25, size: 8, font: fontBold, color: AMBER });
    }

    // ===== SAVE & UPLOAD =====
    const pdfBytes = await pdfDoc.save();
    const fileName = `cr_v${nextVersion}.pdf`;
    const storagePath = `${visit_id}/${fileName}`;

    const { error: uploadErr } = await supabase.storage
      .from('reports')
      .upload(storagePath, pdfBytes, {
        contentType: 'application/pdf',
        upsert: false,
      });
    if (uploadErr) return errorResponse(`Upload failed: ${uploadErr.message}`, 500);

    // GAP-C FIX: public bucket → build public URL directly
    const fileUrl = `${SUPABASE_URL}/storage/v1/object/public/reports/${storagePath}`;

    // Insert generated_documents record
    const { data: docRow, error: docErr } = await supabase
      .from('generated_documents')
      .insert({
        visit_id,
        version: nextVersion,
        format: 'pdf',
        file_url: fileUrl,
        recipients: [],
      })
      .select()
      .single();
    if (docErr) return errorResponse(`DB insert failed: ${docErr.message}`, 500);

    return new Response(
      JSON.stringify({ document_id: docRow.id, file_url: fileUrl, version: nextVersion }),
      { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
    );

  } catch (err: any) {
    console.error('generate-pdf error:', err);
    return errorResponse(err.message || 'Internal error', 500);
  }
});

function errorResponse(message: string, status: number) {
  return new Response(
    JSON.stringify({ error: message }),
    { status, headers: { 'Access-Control-Allow-Origin': '*', 'Content-Type': 'application/json' } }
  );
}
```

```typescript
import { supabase } from '../supabase';
import { GeneratedDocument } from '../types/domain';

const SUPABASE_URL = process.env.EXPO_PUBLIC_SUPABASE_URL || '';

export async function generatePdf(visitId: string): Promise<{ document_id: string; file_url: string; version: number }> {
  const { data: { session } } = await supabase.auth.getSession();
  if (!session) throw new Error('Non authentifié');

  const res = await fetch(`${SUPABASE_URL}/functions/v1/generate-pdf`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${session.access_token}`,
    },
    body: JSON.stringify({ visit_id: visitId }),
  });

  if (!res.ok) {
    const err = await res.json().catch(() => ({ error: 'Erreur inconnue' }));
    throw new Error(err.error || `Erreur ${res.status}`);
  }

  return res.json();
}

export async function sendReport(
  visitId: string,
  documentId: string,
  recipients: string[],
): Promise<{ sent_to: number }> {
  const { data: { session } } = await supabase.auth.getSession();
  if (!session) throw new Error('Non authentifié');

  const res = await fetch(`${SUPABASE_URL}/functions/v1/send-report`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${session.access_token}`,
    },
    body: JSON.stringify({ visit_id: visitId, document_id: documentId, recipients }),
  });

  if (!res.ok) {
    const err = await res.json().catch(() => ({ error: 'Erreur inconnue' }));
    throw new Error(err.error || `Erreur ${res.status}`);
  }

  return res.json();
}

export async function getDocuments(visitId: string): Promise<GeneratedDocument[]> {
  const { data, error } = await supabase
    .from('generated_documents')
    .select('*')
    .eq('visit_id', visitId)
    .order('version', { ascending: false });
  if (error) throw new Error(error.message);
  return data as GeneratedDocument[];
}

export async function getLatestDocument(visitId: string): Promise<GeneratedDocument | null> {
  const { data, error } = await supabase
    .from('generated_documents')
    .select('*')
    .eq('visit_id', visitId)
    .order('version', { ascending: false })
    .limit(1)
    .maybeSingle();
  if (error) throw new Error(error.message);
  return data as GeneratedDocument | null;
}
```

Create `lib/hooks/useDocuments.ts`:

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import * as documentService from '../services/documents';

export function useDocuments(visitId: string) {
  return useQuery({
    queryKey: ['visits', visitId, 'documents'],
    queryFn: () => documentService.getDocuments(visitId),
    enabled: !!visitId,
  });
}

export function useLatestDocument(visitId: string) {
  return useQuery({
    queryKey: ['visits', visitId, 'latest-document'],
    queryFn: () => documentService.getLatestDocument(visitId),
    enabled: !!visitId,
  });
}

export function useGeneratePdf() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: documentService.generatePdf,
    onSuccess: (_data, visitId) => {
      queryClient.invalidateQueries({ queryKey: ['visits', visitId, 'documents'] });
      queryClient.invalidateQueries({ queryKey: ['visits', visitId, 'latest-document'] });
    },
  });
}

export function useSendReport() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: ({ visitId, documentId, recipients }: { visitId: string; documentId: string; recipients: string[] }) =>
      documentService.sendReport(visitId, documentId, recipients),
    onSuccess: (_data, { visitId }) => {
      queryClient.invalidateQueries({ queryKey: ['visits', visitId, 'documents'] });
      queryClient.invalidateQueries({ queryKey: ['visits', visitId, 'latest-document'] });
      queryClient.invalidateQueries({ queryKey: ['visits', visitId] });
    },
  });
}
```
