```typescript
function getDeadlineInfo(deadline: string | null): { label: string; color: string; isLate: boolean; daysLate: number } {
  if (!deadline) return { label: 'Pas d\'échéance', color: colors.textMuted, isLate: false, daysLate: 0 };
  const d = new Date(deadline);
  const now = new Date();
  now.setHours(0, 0, 0, 0);
  const diff = Math.ceil((d.getTime() - now.getTime()) / (1000 * 60 * 60 * 24));
  if (diff < 0) return { label: `En retard · J+${Math.abs(diff)}`, color: '#EF4444', isLate: true, daysLate: Math.abs(diff) };
  if (diff <= 3) return { label: formatDate(deadline), color: '#F59E0B', isLate: false, daysLate: 0 };
  return { label: formatDate(deadline), color: '#10B981', isLate: false, daysLate: 0 };
}
```
