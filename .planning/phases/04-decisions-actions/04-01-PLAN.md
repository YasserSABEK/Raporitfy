```sql
-- ============================================================
-- Phase 04: Decisions & Actions

-- Decisions
CREATE TABLE IF NOT EXISTS decisions (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  visit_id uuid REFERENCES visits(id) ON DELETE CASCADE NOT NULL,
  content text NOT NULL,
  author text NOT NULL,              -- GAP-A: texte libre
  scope text DEFAULT 'chantier_global'
    CHECK (scope IN ('lot_specifique', 'chantier_global', 'contractuel')),
  validated boolean NOT NULL DEFAULT false,
  validated_at timestamptz,          -- GAP-E
  validated_by uuid REFERENCES auth.users(id), -- GAP-E
  created_by uuid REFERENCES auth.users(id) NOT NULL DEFAULT auth.uid(),
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()  -- GAP-B
);

CREATE TRIGGER decisions_updated BEFORE UPDATE ON decisions
  FOR EACH ROW EXECUTE FUNCTION update_timestamp();

CREATE INDEX idx_decisions_visit ON decisions(visit_id);

-- Action Items
CREATE TABLE IF NOT EXISTS action_items (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  visit_id uuid REFERENCES visits(id) ON DELETE CASCADE NOT NULL,
  observation_id uuid REFERENCES observations(id) ON DELETE SET NULL,
  decision_id uuid REFERENCES decisions(id) ON DELETE SET NULL,
  description text NOT NULL,
  owner text NOT NULL,               -- Texte libre (entreprise ou personne)
  deadline date NOT NULL,
  priority text NOT NULL DEFAULT 'moyenne'
    CHECK (priority IN ('basse', 'moyenne', 'haute')),
  status text NOT NULL DEFAULT 'ouverte'
    CHECK (status IN ('ouverte', 'en_cours', 'fermee', 'reportee')),
  created_by uuid REFERENCES auth.users(id) NOT NULL DEFAULT auth.uid(),
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

CREATE TRIGGER action_items_updated BEFORE UPDATE ON action_items
  FOR EACH ROW EXECUTE FUNCTION update_timestamp();

CREATE INDEX idx_action_items_visit ON action_items(visit_id);
CREATE INDEX idx_action_items_status ON action_items(status);
CREATE INDEX idx_action_items_deadline ON action_items(deadline);

-- RLS
ALTER TABLE decisions ENABLE ROW LEVEL SECURITY;
ALTER TABLE action_items ENABLE ROW LEVEL SECURITY;

CREATE POLICY "decisions_org_access" ON decisions FOR ALL USING (
  EXISTS (
    SELECT 1 FROM visits v
    JOIN projects p ON p.id = v.project_id
    JOIN profiles pr ON pr.organization_id = p.organization_id
    WHERE v.id = decisions.visit_id AND pr.id = auth.uid()
  )
);

CREATE POLICY "action_items_org_access" ON action_items FOR ALL USING (
  EXISTS (
    SELECT 1 FROM visits v
    JOIN projects p ON p.id = v.project_id
    JOIN profiles pr ON pr.organization_id = p.organization_id
    WHERE v.id = action_items.visit_id AND pr.id = auth.uid()
  )
);
```

```typescript
import { supabase } from '../supabase';
import { Decision } from '../types/domain';

export async function getDecisions(visitId: string): Promise<Decision[]> {
  const { data, error } = await supabase
    .from('decisions')
    .select('*')
    .eq('visit_id', visitId)
    .order('created_at', { ascending: true });
  if (error) throw new Error(error.message);
  return data as Decision[];
}

export async function createDecision(input: {
  visit_id: string;
  content: string;
  author: string;
  scope?: string;
}): Promise<Decision> {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) throw new Error('Non authentifié');
  const { data, error } = await supabase
    .from('decisions')
    .insert({ ...input, created_by: user.id })
    .select()
    .single();
  if (error) throw new Error(error.message);
  return data as Decision;
}

export async function updateDecision(
  id: string,
  updates: Partial<Pick<Decision, 'content' | 'author' | 'scope' | 'validated'>>
): Promise<Decision> {
  const { data, error } = await supabase
    .from('decisions')
    .update(updates)
    .eq('id', id)
    .select()
    .single();
  if (error) throw new Error(error.message);
  return data as Decision;
}

export async function deleteDecision(id: string): Promise<void> {
  const { error } = await supabase.from('decisions').delete().eq('id', id);
  if (error) throw new Error(error.message);
}
```

```typescript
import { supabase } from '../supabase';
import { ActionItem } from '../types/domain';

export async function getActions(visitId: string): Promise<ActionItem[]> {
  const { data, error } = await supabase
    .from('action_items')
    .select('*')
    .eq('visit_id', visitId)
    .order('deadline', { ascending: true });
  if (error) throw new Error(error.message);
  return data as ActionItem[];
}

// Carry-forward: get ALL open actions for a project (across all visits)
export async function getOpenActionsForProject(projectId: string): Promise<ActionItem[]> {
  const { data, error } = await supabase
    .from('action_items')
    .select('*, visits!inner(project_id, date)')
    .eq('visits.project_id', projectId)
    .in('status', ['ouverte', 'en_cours', 'reportee'])
    .order('deadline', { ascending: true });
  if (error) throw new Error(error.message);
  return data as ActionItem[];
}

// Dashboard: get ALL open actions for the current user's org
export async function getAllOpenActions(): Promise<(ActionItem & { project_name: string; visit_date: string })[]> {
  const { data, error } = await supabase
    .from('action_items')
    .select('*, visits!inner(project_id, date, projects!inner(name))')
    .in('status', ['ouverte', 'en_cours', 'reportee'])
    .order('deadline', { ascending: true });
  if (error) throw new Error(error.message);
  return (data ?? []).map((a: any) => ({
    ...a,
    project_name: a.visits?.projects?.name || '',
    visit_date: a.visits?.date || '',
    visits: undefined,
  }));
}
```

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { getDecisions, createDecision, updateDecision, deleteDecision } from '../services/decisions';

export function useDecisions(visitId: string) {
  return useQuery({
    queryKey: ['visits', visitId, 'decisions'],
    queryFn: () => getDecisions(visitId),
    enabled: !!visitId,
  });
}

export function useCreateDecision() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: createDecision,
    onSuccess: (_data, variables) => {
      qc.invalidateQueries({ queryKey: ['visits', variables.visit_id, 'decisions'] });
    },
  });
}

export function useUpdateDecision() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: ({ id, ...updates }: { id: string } & Parameters<typeof updateDecision>[1]) =>
      updateDecision(id, updates),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['visits'] });
    },
  });
}

export function useDeleteDecision() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: deleteDecision,
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['visits'] });
    },
  });
}
```

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { getActions, getOpenActionsForProject, getAllOpenActions, createAction, updateAction, deleteAction } from '../services/actions';

export function useActions(visitId: string) {
  return useQuery({
    queryKey: ['visits', visitId, 'actions'],
    queryFn: () => getActions(visitId),
    enabled: !!visitId,
  });
}

export function useOpenActionsForProject(projectId: string) {
  return useQuery({
    queryKey: ['projects', projectId, 'open-actions'],
    queryFn: () => getOpenActionsForProject(projectId),
    enabled: !!projectId,
  });
}

export function useAllOpenActions() {
  return useQuery({
    queryKey: ['all-open-actions'],
    queryFn: getAllOpenActions,
  });
}

export function useCreateAction() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: createAction,
    onSuccess: (_data, variables) => {
      qc.invalidateQueries({ queryKey: ['visits', variables.visit_id, 'actions'] });
      qc.invalidateQueries({ queryKey: ['all-open-actions'] });
      qc.invalidateQueries({ queryKey: ['projects'] }); // refresh open-actions for project
    },
  });
}

export function useUpdateAction() {
  const qc = useQueryClient();

    mutationFn: ({ id, ...updates }: { id: string } & Parameters<typeof updateAction>[1]) =>

    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['visits'] });
      qc.invalidateQueries({ queryKey: ['all-open-actions'] });
      qc.invalidateQueries({ queryKey: ['projects'] });
    },
  });
}

export function useDeleteAction() {

    mutationFn: deleteAction,

      qc.invalidateQueries({ queryKey: ['visits'] });
```
