## Task 1: DB Migration — visits, observations, evidence

**[GAP-1]** Added `participants text[]` to visits table.
**[GAP-2]** Added `updated_at` + auto-update trigger to all 3 tables.
**[GAP-3]** Storage bucket `public: false`.

```sql
-- updated_at trigger function
CREATE OR REPLACE FUNCTION update_timestamp()
RETURNS TRIGGER AS $$
BEGIN NEW.updated_at = now(); RETURN NEW; END;
$$ LANGUAGE plpgsql;

-- Visits
CREATE TABLE IF NOT EXISTS visits (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  project_id uuid REFERENCES projects(id) ON DELETE CASCADE NOT NULL,
  date date NOT NULL DEFAULT CURRENT_DATE,
  type text NOT NULL DEFAULT 'chantier'
    CHECK (type IN ('chantier', 'reception', 'levee_reserves')),
  weather text,
  summary text,
  participants text[] DEFAULT '{}',
  status text NOT NULL DEFAULT 'brouillon'
    CHECK (status IN ('brouillon', 'en_revue', 'valide', 'diffuse')),
  version integer NOT NULL DEFAULT 1,
  created_by uuid REFERENCES auth.users(id) NOT NULL DEFAULT auth.uid(),
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);
CREATE TRIGGER visits_updated BEFORE UPDATE ON visits
  FOR EACH ROW EXECUTE FUNCTION update_timestamp();

-- Observations
CREATE TABLE IF NOT EXISTS observations (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  visit_id uuid REFERENCES visits(id) ON DELETE CASCADE NOT NULL,
  lot text NOT NULL DEFAULT '',
  zone text NOT NULL DEFAULT '',
  description text NOT NULL,
  severity text NOT NULL DEFAULT 'mineur'
    CHECK (severity IN ('mineur', 'majeur', 'critique')),
  classification text NOT NULL DEFAULT 'constat',
  source text,
  confidence numeric NOT NULL DEFAULT 1.0,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);
CREATE TRIGGER observations_updated BEFORE UPDATE ON observations
  FOR EACH ROW EXECUTE FUNCTION update_timestamp();

-- Evidence
CREATE TABLE IF NOT EXISTS evidence (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  observation_id uuid REFERENCES observations(id) ON DELETE CASCADE NOT NULL,
  type text NOT NULL DEFAULT 'photo'
    CHECK (type IN ('photo', 'audio', 'document', 'note')),
  file_url text,
  content text,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);
CREATE TRIGGER evidence_updated BEFORE UPDATE ON evidence
  FOR EACH ROW EXECUTE FUNCTION update_timestamp();

-- Indexes
CREATE INDEX idx_visits_project ON visits(project_id, date DESC);
CREATE INDEX idx_observations_visit ON observations(visit_id);
CREATE INDEX idx_evidence_observation ON evidence(observation_id);

-- RLS
ALTER TABLE visits ENABLE ROW LEVEL SECURITY;
ALTER TABLE observations ENABLE ROW LEVEL SECURITY;
ALTER TABLE evidence ENABLE ROW LEVEL SECURITY;

CREATE POLICY "visits_org_access" ON visits FOR ALL USING (
  EXISTS (
    SELECT 1 FROM projects p
    JOIN profiles pr ON pr.organization_id = p.organization_id
    WHERE p.id = visits.project_id AND pr.id = auth.uid()
  )
);

CREATE POLICY "observations_via_visit" ON observations FOR ALL USING (
  EXISTS (
    SELECT 1 FROM visits v
    JOIN projects p ON p.id = v.project_id
    JOIN profiles pr ON pr.organization_id = p.organization_id
    WHERE v.id = observations.visit_id AND pr.id = auth.uid()
  )
);

CREATE POLICY "evidence_via_observation" ON evidence FOR ALL USING (
  EXISTS (
    SELECT 1 FROM observations o
    JOIN visits v ON v.id = o.visit_id
    JOIN projects p ON p.id = v.project_id
    JOIN profiles pr ON pr.organization_id = p.organization_id
    WHERE o.id = evidence.observation_id AND pr.id = auth.uid()
  )
);

-- Storage bucket: PRIVATE (signed URLs for access)
INSERT INTO storage.buckets (id, name, public)
VALUES ('evidence-photos', 'evidence-photos', false)
ON CONFLICT (id) DO NOTHING;

CREATE POLICY "evidence_upload" ON storage.objects FOR INSERT
  WITH CHECK (bucket_id = 'evidence-photos' AND auth.role() = 'authenticated');
CREATE POLICY "evidence_read" ON storage.objects FOR SELECT
  USING (bucket_id = 'evidence-photos' AND auth.role() = 'authenticated');
```

**Verify:** `SELECT tablename FROM pg_tables WHERE schemaname='public' AND tablename IN ('visits','observations','evidence');` → 3 rows
