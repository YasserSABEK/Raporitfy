```typescript
async function uploadPhotosParallel(uris: string[], maxConcurrent = 3) {
  const results: string[] = [];
  for (let i = 0; i < uris.length; i += maxConcurrent) {
    const batch = uris.slice(i, i + maxConcurrent);
    const urls = await Promise.allSettled(
      batch.map(uri => uploadPhoto(uri))
    );
    urls.forEach(r => {
      if (r.status === 'fulfilled') results.push(r.value);
    });
    setUploadProgress(Math.min(i + maxConcurrent, uris.length));
  }
  return results;
}
```
3. Create evidence records for each successful upload
